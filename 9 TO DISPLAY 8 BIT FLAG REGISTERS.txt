9 TO DISPLAY 8 BIT FLAG REGISTERS :

DATA SEGMENT
msg DB 0DH,0AH,'Flag Register Bits: $'
newline DB 0DH,0AH,'$'
flag DW ?
DATA ENDS

CODE SEGMENT
ASSUME CS:CODE, DS:DATA

START:
MOV AX, DATA
MOV DS, AX

MOV DX, OFFSET msg
MOV AH, 09H
INT 21H

MOV DX, OFFSET newline
MOV AH, 09H
INT 21H

CLI
STC
STD

PUSHF
POP BX
MOV flag, BX

MOV CX, 16
MOV BX, 8000H

AGAIN:
MOV AX, flag
AND AX, BX
JZ ZERO
MOV DL, '1'
MOV AH, 02H
INT 21H
JMP SPACE

ZERO:
MOV DL, '0'
MOV AH, 02H
INT 21H

SPACE:
MOV DL, ' '
MOV AH, 02H
INT 21H

ROR BX, 1
LOOP AGAIN

MOV AH, 4CH
INT 21H

CODE ENDS
END START
